Prepping datasets

Load from previously saved datasets

rm(list=ls())

## Load all required libraries
library(utils)       # for read.csv
library(lpSolveAPI)  # for lp optimisation
library(dplyr)       # for piping %>%, mutate()
library(reshape2)    # for reshaping/melting datasets
library(ggplot2)     # for plotting

## Load data, use utils::read.csv() due to player european name
# my_team    = read.csv("./Output/faizmk_base_team.csv")
dfsummary   = read.csv("../Data/week03_01summary.csv")
dfsumm_prev = read.csv("../Data/week02_01summary.csv")
dfhistory   = read.csv("../Data/week03_03history_.csv")
team_name   = read.csv("../Data/week03_05teamsummary.csv")

## Prep dfsummary - Merge (to include previous ep_next)
dfsummary$ep_next_prev = dfsumm_prev$ep_next[match(dfsummary$id, dfsumm_prev$id)]
dfsummary = mutate(dfsummary,
                   ep_this_prev = dfsumm_prev$ep_this[match(dfsummary$id, dfsumm_prev$id)],
                   ep_next_prev = dfsumm_prev$ep_next[match(dfsummary$id, dfsumm_prev$id)],
                   event_points_prev = dfsumm_prev$event_points[match(dfsummary$id, dfsumm_prev$id)],
                   total_points_prev = dfsumm_prev$total_points[match(dfsummary$id, dfsumm_prev$id)])

## Prep dfhistory - Cleanup & Merge (to include team strength)
dfhistory$event_points  = dfsummary$event_points[match(dfhistory$player_id, dfsummary$id)]
dfhistory$opponent_team = team_name$short_name[match(dfhistory$opponent_team, team_name$id)]
dfhistory$player_team   = dfsummary$team[match(dfhistory$player_id, dfsummary$id)]
# merge opponent strength
dfhistory = merge(dfhistory, team_name[,c("short_name", "strength")], 
                  by.x="opponent_team", by.y="short_name")
names(dfhistory)[grep("^strength", names(dfhistory))] = "opponent_strength"
# merge player team strength
dfhistory = merge(dfhistory, team_name[,c("short_name", "strength")], 
                       by.x="player_team", by.y="short_name")
names(dfhistory)[grep("^strength", names(dfhistory))] = "team_strength"

Fixtures Predictive Power

Here we scrutinize the data to see how Team Strength (own’s & opponent’s) affect the distribution of (1) gameweek points as well as influencing (2) yellow & red cards.

– On Individual Points Gain

Analysis on distribution of gameweek points based on team’s strength, either own’s or opponent’s or a combination.

dfhistory$opponent_strength = as.factor(dfhistory$opponent_strength)
dfhistory$team_strength     = as.factor(dfhistory$team_strength)

## Plot individual points (season's points so far), segregated by opponent's team strength
p = ggplot(dfhistory) + 
  geom_boxplot(aes(x=opponent_strength, y=event_points)) + 
  ggtitle("Individual Points Against Opponent Strength", subtitle = "Overall")
print(p)

## Plot individual points (season's points so far), segregated by own's team strength
q = ggplot(dfhistory) + 
  geom_boxplot(aes(x=team_strength, y=event_points)) + 
  ggtitle("Individual Points Breakdown by Own's Team Strength", subtitle = "Overall")
print(q)

## Now let's see segregation of individual points, segregated by opponent's team strength, IF we breakdown
## by own's team strength.
p = p + facet_grid(. ~ team_strength) + 
  ggtitle("Individual Points Against Opponent Strength", subtitle = "Faceted by Own Team Strength")
print(p)

– Using element_type key indicator

Analysis on which player position (element_type) will statistically give us better chance of getting more points

df = dfsummary[!dfsummary$element_type=="GKP",
               c("web_name", "element_type", "influence", "ict_index", "minutes", "event_points")]
## Gameweek Elements by Position
dfm = melt(df, id.vars=c("web_name", "element_type"))
p = ggplot(data=dfm) + geom_boxplot(aes(x=reorder(element_type, value, FUN=function(x){1-mean(x)}), y=value)) +
  ggtitle("Gameweek Elements by Position", subtitle="Faceted by Elements") + xlab("Position / element_type") +
  facet_wrap(~ variable, scales="free")
print(p)

## Gameweek Points by Position
p = ggplot(data=df) + geom_violin(aes(x=reorder(element_type, event_points, FUN=mean), event_points)) +
  ggtitle("Gameweek Points by Position", subtitle="Overall") + xlab("Position / element_type")
print(p)
